
# CREATE REPOSITORY AND PUSH

# initialize local directory as git repository
git init
# add local files
git add .
# commit files
git commit -m "First commit"

# create repository on github and copy URL

# link local to remote URL
git remote add origin https://github.com/lakemike/debian-duplicity-rclone
git remote -v

# push to main
git push origin main

# or using branches:
# name local branch:
git branch -m <new_name>
# push to add to this branc
git push origin -u <new_name>

use developer access token as password!

# PULL FROM REPOSITORY
git clone <repository_clone_path>

# MAKE CHANGES AND COMMIT
git commit -m "Second commit"
git push origin main

# on grautvornix (treats current directory as /local, does not have any global config?)
